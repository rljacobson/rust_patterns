1/*!
2# Typed and Type-Erased API
3
4An illustration of a type, a database index, having both a typed and type-erased API.
5
6Sometimes an implementation needs access to a concrete type while providing a type-erased interface. In some situations,
7a type-erased interface can provide the same functionality as a typed-API but at some cost, and so you want to provide
8both a typed and type-erased interface.
9
10In this context, by _typed_ I mean that a concrete type is known at compile time. Generally
11a typed API will be _generic_ over the type but monomorphized at compile time. (Stop
12worrying about how monomorphization increases the size of the final binary. And, while
13you're at it, stop obsessing about string copies.) The concrete type is `Index<T>` (notice
14the generic `T`). The _typed_ API is just the methods provided by `impl<T> Index<T>{...}`.
15
16A type-erased API, on the other hand, will be monomorphic over a trait object. We define the type-erased API via the
17trait `TypeErasedIndex` (notice the lack of a generic `T`) and provide a _blanket implementation_ for all `Index<T>`s.
18
19Notice that it turns out to be convenient to implement many of the methods of the typed
20API by deferring to the type-erased API, which is typical. [You will want to do this when
21you can to keep your code DRY.](https://en.wikipedia.org/wiki/Don%27t_repeat_yourself)
22Don't be a slave to the DRY principle; violate it whenever it makes
23sense. These rules of thumb are not an excuse to turn off your brain.
24
25*/

27use std::{any::Any, collections::HashSet, hash::Hash};
28
29use hashbrown::{HashTable, hash_table::OccupiedEntry};
30
31use crate::hashing::one_shot_128;
32
33/// A "boxed" `TypeErasedIndex`, use anywhere you need a type-erased `Index<T>`
34pub type BxIndex = Box<dyn TypeErasedIndex>;
35
36pub type EntityId = u64;
37pub type HashValue = u128;
38
39/// The typed `Index<T>`
40#[derive(Default)]
41pub struct Index<T: Hash + Eq + Clone + Any> {
42    /// We store a copy of the value here so that we can iterate over it in the typed API, and so that the type-erased
43    /// API can access some serialization of it.
44    lookup: HashTable<(T, HashSet<EntityId>)>,
45}
46
47/// Contains the typed API
48impl<T: Hash + Eq + Clone + Any> Index<T> {
49    pub fn new() -> Self {
50        Self {
51            lookup: HashTable::default(),
52        }
53    }
54
55    /// Inserts an entity into the set associated with `key`, creating a new set if one does
56    /// not yet exist. Returns a `bool` according to whether the `entity_id` already existed
57    /// in the set. Observe that several of these just defer to the untyped implementation.
58    pub fn insert_entity(&mut self, key: &T, entity_id: EntityId) -> bool {
59        let hash = one_shot_128(&key);
60
61        // `hasher` is called if entries need to be moved or copied to a new table.
62        // This must return the same hash value that each entry was inserted with.
63        let hasher = |(stored_value, _stored_set): &_| one_shot_128(stored_value) as u64;
64
65        // Equality is determined by comparing the full 128-bit hashes. We do not expect any collisions before the heat
66        // death of the universe.
67        let hash128_equality = |(stored_value, _): &_| one_shot_128(stored_value) == hash;
68
69        self.lookup
70            .entry(hash as u64, hash128_equality, hasher)
71            .or_insert_with(|| (key.clone(), HashSet::new()))
72            .get_mut()
73            .1
74            .insert(entity_id)
75    }
76
77    /// Inserting a new _value_ requires the value itself.
78    pub fn insert_value(
79        &mut self,
80        key: T,
81        set: HashSet<EntityId>,
82    ) -> OccupiedEntry<'_, (T, HashSet<EntityId>)> {
83        let hash = one_shot_128(&key);
84        // `hasher` is called if entries need to be moved or copied to a new table.
85        // This must return the same hash value that each entry was inserted with.
86        let hasher = |(stored_value, _stored_set): &_| one_shot_128(stored_value) as u64;
87        self.lookup.insert_unique(hash as u64, (key, set), hasher)
88    }
89
90    /// Gets an immutable reference to the set associated with the `key` if it exists. Observe that we just defer to
91    /// the untyped implementation.
92    pub fn get(&self, key: &T) -> Option<&HashSet<EntityId>> {
93        let hash = one_shot_128(&key);
94        self.get_with_hash(hash)
95    }
96
97    /// Gets a mutable reference to the set associated with the `key` if it exists. Observe that we just defer to
98    //   /// the untyped implementation.
99    pub fn get_mut(&mut self, key: &T) -> Option<&mut HashSet<EntityId>> {
100        let hash = one_shot_128(&key);
101        self.get_with_hash_mut(hash)
102    }
103
104    // Possibly other methods ...
105}
106
107/// This trait Encapsulates the type-erased API.
108pub trait TypeErasedIndex {
109    /// Inserting a new entity only requires the hash but requires the set associated with the hash to already exist.
110    ///
111    /// If the set corresponding to the hash exists, inserts the `entity_id` into the associated set, returning a `bool`
112    /// according to whether the `entity_id` was already in the set.
113    /// If the set does not exist, returns `Err(())`
114    fn insert_entity_with_hash(&mut self, hash: HashValue, entity_id: EntityId)
115    -> Result<bool, ()>;
116
117    /// Fetching a set only requires the hash.
118    fn get_with_hash(&self, hash: HashValue) -> Option<&HashSet<EntityId>>;
119
120    /// Fetching a set only requires the hash.
121    fn get_with_hash_mut(&mut self, hash: HashValue) -> Option<&mut HashSet<EntityId>>;
122
123    /// Does the index contain the given hash?
124    fn has_hash(&self, hash: HashValue) -> bool;
125}
126
127/// A blanket implementation of the type-erased API for all `Index<T>`s.
128impl<T: Hash + Eq + Clone + Any> TypeErasedIndex for Index<T> {
129    fn insert_entity_with_hash(
130        &mut self,
131        hash: HashValue,
132        entity_id: EntityId,
133    ) -> Result<bool, ()> {
134        // Equality is determined by comparing the full 128-bit hashes. We do not expect any collisions before the heat
135        // death of the universe.
136        let hash128_equality = |(stored_value, _): &_| one_shot_128(stored_value) == hash;
137
138        let entities = self
139            .lookup
140            .find_mut(hash as u64, hash128_equality)
141            .map(|(_, set)| set)
142            .ok_or(())?;
143        Ok(entities.insert(entity_id))
144    }
145
146    fn get_with_hash(&self, hash: HashValue) -> Option<&HashSet<EntityId>> {
147        // Equality is determined by comparing the full 128-bit hashes. We do not expect any collisions before the heat
148        // death of the universe.
149        let hash128_equality = |(stored_value, _): &_| one_shot_128(stored_value) == hash;
150        self.lookup
151            .find(hash as u64, hash128_equality)
152            .map(|(_, set)| set)
153    }
154
155    fn get_with_hash_mut(&mut self, hash: HashValue) -> Option<&mut HashSet<EntityId>> {
156        // Equality is determined by comparing the full 128-bit hashes. We do not expect any collisions before the heat
157        // death of the universe.
158        let hash128_equality = |(stored_value, _): &_| one_shot_128(stored_value) == hash;
159        self.lookup
160            .find_mut(hash as u64, hash128_equality)
161            .map(|(_, set)| set)
162    }
163
164    fn has_hash(&self, hash: HashValue) -> bool {
165        self.get_with_hash(hash).is_some()
166    }
167}
